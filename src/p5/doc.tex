	\Function{insertafter}{Node $node$, Node $newNode$}
		\If{$node = \phi$}

		\Else
			\State $newNode.next = node.next$
			\State $node.next = newNode$
		\EndIf
	\EndFunction

\subsection{Traversal}

\begin{algorithmic}
	\Function{iterate}{Node $node$}
		\If{$node \neq \phi$}
			\State $iter \gets node$
			\Repeat
				\State \Call{foo}{node.data}
				\State $node = node.next$
			\Until{$iter \neq node$}
		\EndIf
	\EndFunction
\end{algorithmic}


\subsection{Searching}

\begin{algorithmic}
	\Function{search}{Node $node$, data $x$}
		\State $found \gets false$
		\If{$node \neq \phi$}
			\State $iter \gets node$
			\Repeat
				\If{$node.data = x$}
					\State $found = true$
				\EndIf
				\State $node = node.next$
			\Until{$iter \neq node$ \textbf{and} $found$ \textbf{not} $true$}
		\EndIf
		\State \Return $\phi$
	\EndFunction
\end{algorithmic}
