\chapter{Singly Linked List}

\section{Aim}

To implement linked list operations on a "Singly Linked List".

\section{Algorithm}

Let the $node$ data structure contain two fields,
data and the reference to the next node. Let there
also be a variable $firstNode$ to keep track of the
first node in the list, which is $\phi$ when the list
is empty.

\begin{lstlisting}[style=myC]
record Node
{
	data;
	Node next;
}
record List {
	Node firstNode;
}

\end{lstlisting}

\subsection{Insertion}

\begin{algorithmic}
	\Function{insertBeginning}{List $list$, Node $newNode$}
		\State $newNode.next = list.firstNode$
		\State $list.firstNode = newNode$
	\EndFunction

	\Function{insertAfter}{Node $node$, Node $newNode$}
		\State $newNode.next = node.next$
		\State $node.next = newNode$
	\EndFunction
\end{algorithmic}

\subsection{Deletion}

\begin{algorithmic}
	\Function{deleteBeginning}{List $list$}
		\State $temp \gets list.firstNode$
		\State $list.firstNode = temp.next$
		\State \Call{destroy}{temp}
	\EndFunction
	
	\Function{deleteAfter}{Node $node$}
		\State $temp \gets node.next$
		\State $node.next = temp.next$
		\State \Call{destroy}{temp}
	\EndFunction
\end{algorithmic}

\subsection{Traversal}

\begin{algorithmic}
	\Function{traverse}{List $list$}
		\State $node \gets list.firstNode$
		\While{$node$ \textbf{not} $\phi$}
			\State \Call{foo}{$node.data$}
			\State $node = node.next$
		\EndWhile
	\EndFunction
\end{algorithmic}

\subsection{Searching}

\begin{algorithmic}
	\Function{search}{data $x$, List $list$}
		\State $found \gets false$
		\State $node \gets list.firstNode$
		\While{$node$ \textbf{not} $\phi$ \textbf{and} $found$ \textbf{not} $true$ }
			\If{$node.data = x$}
				\State $found = true$
			\EndIf
		\EndWhile
		\State \Return{$found$}
	\EndFunction
\end{algorithmic}
